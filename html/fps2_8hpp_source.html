208 [[nodiscard]] MFPS inv(int d) const {
209 // 参考：https://nyaannyaan.github.io/library/fps/formal-power-series.hpp
210 // verify : https://judge.yosupo.jp/problem/inv_of_formal_power_series
211
212 // 【方法】
213 // 1 / f mod z^d を求めることは，
214 // f g = 1 (mod z^d)
215 // なる g を求めることである．
216 // この d の部分を 1, 2, 4, ..., 2^i と倍々にして求めていく．
217 //
218 // d = 1 のときについては
219 // g = 1 / f[0] (mod z^1)
220 // である．
221 //
222 // 次に，
223 // g = h (mod z^k)
224 // が求まっているとして
225 // g mod z^(2 k)
226 // を求める．最初の式を変形していくことで
227 // g - h = 0 (mod z^k)
228 // ⇒ (g - h)^2 = 0 (mod z^(2 k))
229 // ⇔ g^2 - 2 g h + h^2 = 0 (mod z^(2 k))
230 // ⇒ f g^2 - 2 f g h + f h^2 = 0 (mod z^(2 k))
231 // ⇔ g - 2 h + f h^2 = 0 (mod z^(2 k)) 　(f g = 1 (mod z^d) より)
232 // ⇔ g = (2 - f h) h (mod z^(2 k))
233 // を得る．
234 //
235 // この手順を d ≦ 2^i となる i まで繰り返し，d 次以上の項を削除すればよい．
236
237 Assert(!c.empty());
238 Assert(c[0] != 0);
239
240 MFPS g(c[0].inv());
241 for (int k = 1; k < d; k <<= 1) {
242 int len = max(min(2 * k, d), 1);
243 MFPS tmp(0, len);
244 rep(i, min(len, n)) tmp[i] = -c[i]; // -f
245 tmp *= g; // -f h
246 tmp.resize(len);
247 tmp[0] += 2; // 2 - f h
248 g *= tmp; // (2 - f h) h
249 g.resize(len);
250 }
251
252 return g;
253 }

258 [[nodiscard]] MFPS quotient(const MFPS& g) const {
259 // 参考 : https://nyaannyaan.github.io/library/fps/formal-power-series.hpp
260 // verify : https://judge.yosupo.jp/problem/division_of_polynomials
261
262 // 【方法】
263 // f(x) = g(x) q(x) + r(x) となる q(x) を求める．
264 // f の次数は n-1, g の次数は m-1 とする．(n ≧ m)
265 // 従って q の次数は n-m，r の次数は m-2 となる．
266 //
267 // f^R で f の係数列を逆順にした多項式を表す．すなわち
268 // f^R(x) := f(1/x) x^(n-1)
269 // である．他の多項式も同様とする．
270 //
271 // 最初の式で x → 1/x と置き換えると，
272 // f(1/x) = g(1/x) q(1/x) + r(1/x)
273 // ⇔ f(1/x) x^(n-1) = g(1/x) q(1/x) x^(n-1) + r(1/x) x^(n-1)
274 // ⇔ f(1/x) x^(n-1) = g(1/x) x^(m-1) q(1/x) x^(n-m) + r(1/x) x^(m-2) x^(n-m+1)
275 // ⇔ f^R(x) = g^R(x) q^R(x) + r^R(x) x^(n-m+1)
276 // ⇒ f^R(x) = g^R(x) q^R(x) (mod x^(n-m+1))
277 // ⇒ q^R(x) = f^R(x) / g^R(x) (mod x^(n-m+1))
278 // を得る．
279 //
280 // これで q を mod x^(n-m+1) で正しく求めることができることになるが，
281 // q の次数は n-m であったから，q 自身を正しく求めることができた．
282
283 if (n < g.n) return MFPS();
284 return ((this->rev() / g.rev()).resize(n - g.n + 1)).rev();
285 }
