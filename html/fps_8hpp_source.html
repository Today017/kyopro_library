131 Fps Pow(Fps f, ll M, int len=-1) {

134 if(M==0) {
135 res[0]=1;
136 return res;
137 }

139 if(f[i]==0) continue;

145 if(i) len-=i*M;
146 g=Log(g,len);
147 for(Mod998& x:g) x*=M;
148 g=Exp(g,len);

150 Mod998 c=1;

154 M>>=1;

163 pair<Fps,Fps> Div(Fps f, Fps g) {
