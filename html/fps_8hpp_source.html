40 /// @brief 多項式 f に対し、f*g = 1 なる g を返す
41 /// @note O(N log(N))
42 /// @ref https://judge.yosupo.jp/problem/inv_of_formal_power_series
43 Poly Inv(Poly f, int len=-1) {
44 /**
45 * 方程式 h(g) = 1/g - f = 0 の解を求める問題に変換する
46 * ニュートン法より
47 * g_{n+1} = g_n - h(g_n)/h'(g_n)
48 * h'(g) = - 1/g^2 より
49 * g_{n+1} = g_n - (1/g_n - f) / (-1/g_n^2)
50 * = g_n + g_n - f*g_n^2
51 * = g_n * (2 - f*g_n)
52 * ここで、f*g_n を上位次数の項と下位次数の項 fg_low, fg_high に分けると、
53 * fg_low + fg_high = f*g_n, fg_low = 1 (mod x^d)
54 * よって、
55 * g_{n+1} = g_n * (2 - fg_low - fg_high)
56 * = g_n * (1 - fg_high)
57 * = g_n - g_n*fg_high
58 */
59
60 if(len==-1) len=f.size();
61 assert(f[0]!=0);
62 Poly g={f[0].inv()};
63 int s=1;
64 while(s<len) {
65 Poly nxtg(s*2,0), res(s*2,0);
66 g.resize(s*2);
67 for(int i=0; i<s*2; i++) {
68 if(int(f.size())>i) res[i]=f[i];
69 nxtg[i]=g[i];
70 }
71
72 //fg_high を計算
73 NTT998(g); NTT998(res);
74 for(int i=0; i<s*2; i++) res[i]*=g[i];
75 NTT998(res,true); for(int i=0; i<s; i++) res[i]=0; //fg_high
76
77 //fg_high * g_n を計算
78 NTT998(res); for(int i=0; i<s*2; i++) res[i]*=g[i]; NTT998(res,true);
79
80 for(int i=s; i<s*2; i++) nxtg[i]-=res[i];
81 swap(nxtg,g);
82 s*=2;
83 }
84 g.resize(len);
85 return g;
86 }

111 pair<Poly,Poly> Div(Poly f, Poly g) {

115 reverse(all(f)); reverse(all(g));
116 f.resize(n-m+1);

216 /// @brief 多項式 f について、e^f = Σ[k=0~len-1](f(x)^k/k!) を返す
217 Poly Exp(Poly f, int len=-1) {
218 /**
219 * 方程式 h(g) = log(g) - f = 0 の解を求める問題に変換する。
220 * ニュートン法より
221 * g_{n+1} = g_n - h(g_n)/h'(g_n)
222 * h(g) = log(g) - f, h'(g) = 1/g より
223 * g_{n+1} = g_n - (log(g_n) - f) / (1/g_n)
224 * = g_n - g_n(log(g_n) - f)
225 * = g_n * (1 - log(g_n) + f)
226 * ここで、log(g_n) を上位項と下位項 lg_low, lg_high に分けると、
227 * log(g_n) = lg_low + lg_high, lg_low = f_low
228 * よって、
229 * g_{n+1} = g_n(1 - f_low - lg_high + f)
230 * = g_n(1 + (f - log(g))_high)
231 * = g_n + g_n * (f - log(g))_high
232 */
233
234 if(len==-1) len=f.size();
235 if(len==0) return{};
236 if(len==1) return{Mod998(1)};
237 assert(!f.empty() && f[0]==0);
238
239 int s=1;
240 Poly g={Mod998(1)};
241
242 while(s<len) {
243 Poly A=g,B=g;
244 A=Differential(A); //A = g'
245 B=Inv(B,2*s); //B = 1/g
246 A.resize(2*s);
247 A=Internal::CyclicConvolution(A,B); //A = g'/g
248 A.pop_back();
249 A=Integral(A); //A = ∫(g'/g)dx = log(g)
250
251 //A = (f-log(g_n))_high
252 for(int i=0; i<s; i++) A[i]=0;
253 for(int i=s; i<s*2; i++) A[i]=(i<(int)f.size() ? f[i] : 0)-A[i];
254
255 //g_{n+1} = g_n + g_n * (f-log(g))_high
256 g.resize(2*s);
257 B=Internal::CyclicConvolution(A,g);
258 for(int i=s; i<s*2; i++) g[i]=B[i];
259 s*=2;
260 }
261 g.resize(len);
262 return g;
263 }
