39 template<typename T>
40 vector<T> Inv(vector<T> f, int len=-1) {
41 if(len==-1) len=f.size();
42 assert(f[0]!=0);
43 vector<T> g={1/f[0]};
44 int s=1;
45 while(s<len) {
46 //g=2g_s-f(g_s)^2(mod x^(2*s))
47 //g=g-(fg-1)g
48 //(fg-1)=0(mod x^(s))
49 vector<T> n_g(s*2,0),f_s(s*2,0);
50 g.resize(s*2);
51 for(int i=0; i<s*2; i++) {
52 if(int(f.size())>i) f_s[i]=f[i];
53 n_g[i]=g[i];
54 }
55 atcoder::internal::butterfly(g);
56 atcoder::internal::butterfly(f_s);
57 for(int i=0; i<s*2; i++) f_s[i]*=g[i];
58 atcoder::internal::butterfly_inv(f_s);
59 T iz=1/(T)(s*2);
60 for(int i=s; i<s*2; i++) f_s[i]*=iz;
61 for(int i=0; i<s; i++) f_s[i]=0;
62 atcoder::internal::butterfly(f_s);
63 for(int i=0; i<s*2; i++) f_s[i]*=g[i];
64 atcoder::internal::butterfly_inv(f_s);
65 for(int i=s; i<s*2; i++) n_g[i]-=f_s[i]*iz;
66 swap(n_g,g);
67 s*=2;
68 }
