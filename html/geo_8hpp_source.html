237 if(n<=1)return points;
238 sort(points.begin(),points.end(),[](const Point&l,const Point&r)->bool{
239 if(almostEqual(l.y,r.y))return lessThan(l.x,r.x);
240 return lessThan(l.y,r.y);

244 for(int i=2;i<n;++i){
245 while((int)upper.size()>=2&&ccw(upper.end()[-2],upper.end()[-1],points[i])!=CLOCKWISE){
246 if(ccw(upper.end()[-2],upper.end()[-1],points[i])==ONLINE_FRONT&&include_collinear)break;
247 upper.pop_back();
248 }
249 upper.push_back(points[i]);

251 if(ccw(lower.end()[-2],lower.end()[-1],points[i])==ONLINE_FRONT&&include_collinear)break;

257 upper.pop_back();

341 sort(segments.begin(),segments.end(),[](const Segment&lhs,const Segment&rhs)->bool{
342 return lessThan(min(lhs.a.x,lhs.b.x),min(rhs.a.x,rhs.b.x));
343 });
344 for(const auto&seg:segments){
345 if(seg.a.y==seg.b.y){
346 //Horizontal segment
347 Real y=seg.a.y;
348 Real x1=min(seg.a.x,seg.b.x);
349 Real x2=max(seg.a.x,seg.b.x);
350 events.emplace_back(x1,0,y,y);
351 events.emplace_back(x2,2,y,y);
352 }else{
353 //Vertical segment
354 Real x=seg.a.x;
355 Real y1=min(seg.a.y,seg.b.y);
356 Real y2=max(seg.a.y,seg.b.y);
357 events.emplace_back(x,1,y1,y2);
358 }
359 }
360 sort(events.begin(),events.end());
361 set<Real>activeSegments;
362 int intersectionCount=0;
363 for(const auto&event:events){
364 if(event.type==0){
365 //Add horizontal segment to active set
366 activeSegments.insert(event.y1);
367 }else if(event.type==2){
368 //Remove horizontal segment from active set
369 activeSegments.erase(event.y1);
370 }else if(event.type==1){
371 //Count intersections with vertical segment
372 auto lower=activeSegments.lower_bound(event.y1);
373 auto upper=activeSegments.upper_bound(event.y2);
374 intersectionCount+=distance(lower,upper);
375 }
376 }
377 return intersectionCount;
