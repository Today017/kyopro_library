105 /// @brief 3点 p0, p1, p2 の進行方向を返す
106 Orientation ccw(const Point& p0, const Point& p1, const Point& p2) {
107 Point a=p1-p0;
108 Point b=p2-p0;
109 Real cross_product=a.cross(b);
110 if(greaterThan(cross_product,0)) return COUNTER_CLOCKWISE;
111 if(lessThan(cross_product,0)) return CLOCKWISE;
112 if(lessThan(a.dot(b),0)) return ONLINE_BACK;
113 if(lessThan(a.norm(),b.norm())) return ONLINE_FRONT;
114 return ON_SEGMENT;
115 }

283 /// @brief 凸包を求める
284 vector<Point> convexHull(vector<Point>& points, bool include_collinear=false) {
285 int n=points.size();
286 if(n<=1) return points;
287 sort(points.begin(), points.end(), [](const Point& l, const Point& r)-> bool {
288 if(almostEqual(l.y,r.y)) return lessThan(l.x,r.x);
289 return lessThan(l.y,r.y);
290 });
291 if(n==2) return points;
292 vector<Point> upper={points[0],points[1]},lower={points[0],points[1]};
293 for(int i=2; i<n; i++) {
294 while(upper.size()>=2 && ccw(upper.end()[-2],upper.end()[-1],points[i])!=CLOCKWISE) {
295 if(ccw(upper.end()[-2],upper.end()[-1],points[i])==ONLINE_FRONT && include_collinear) break;
296 upper.pop_back();
297 }
298 upper.push_back(points[i]);
299 while(lower.size()>=2 && ccw(lower.end()[-2],lower.end()[-1],points[i])!=COUNTER_CLOCKWISE) {
300 if(ccw(lower.end()[-2],lower.end()[-1],points[i])==ONLINE_FRONT && include_collinear) break;
301 lower.pop_back();
302 }
303 lower.push_back(points[i]);
304 }
305 reverse(upper.begin(),upper.end());
306 upper.pop_back(); lower.pop_back();
307 lower.insert(lower.end(),upper.begin(),upper.end());
308 return lower;
309 }

346 }

380 /// @brief 線分の交差数を数える
381 int countIntersections(vector<Segment> segments) {
382 struct Event {
383 Real x;
384 int type;//0:horizontal start,1:vertical,2:horizontal end
385 Real y1,y2;
386 Event(Real x, int type, Real y1, Real y2): x(x), type(type), y1(y1), y2(y2){}
387 bool operator<(const Event& other) const {
388 if(x==other.x) return type<other.type;
389 return x<other.x;
390 }
391 };
392 vector<Event> events;
393 sort(segments.begin(),segments.end(),[](const Segment& lhs, const Segment& rhs)-> bool {
394 return lessThan(min(lhs.a.x,lhs.b.x),min(rhs.a.x,rhs.b.x));
395 });
396 for(const auto& seg: segments) {
397 if(seg.a.y==seg.b.y) {
398 //Horizontal segment
399 Real y=seg.a.y;
400 Real x1=min(seg.a.x,seg.b.x);
401 Real x2=max(seg.a.x,seg.b.x);
402 events.emplace_back(x1,0,y,y);
403 events.emplace_back(x2,2,y,y);
404 } else {
405 //Vertical segment
406 Real x=seg.a.x;
407 Real y1=min(seg.a.y,seg.b.y);
408 Real y2=max(seg.a.y,seg.b.y);
409 events.emplace_back(x,1,y1,y2);
410 }
411 }
412 sort(events.begin(),events.end());
413 set<Real> activeSegments;
414 int intersectionCount=0;
415 for(const auto& event: events){
416 if(event.type==0) {
417 //Add horizontal segment to active set
418 activeSegments.insert(event.y1);
419 } else if(event.type==2) {
420 //Remove horizontal segment from active set
421 activeSegments.erase(event.y1);
422 } else if(event.type==1) {
423 //Count intersections with vertical segment
424 auto lower=activeSegments.lower_bound(event.y1);
425 auto upper=activeSegments.upper_bound(event.y2);
426 intersectionCount+=distance(lower,upper);
427 }
428 }
429 return intersectionCount;
430 }

470 /// @brief 円と直線の交点を求める
471 vector<Point> getCircleLineIntersection(const Circle& c, Point p1, Point p2) {
472 Real cx=c.center.x,cy=c.center.y,r=c.r;
473 Real dx=p2.x-p1.x;
474 Real dy=p2.y-p1.y;
475 Real a=dx*dx+dy*dy;
476 Real b=2*(dx*(p1.x-cx)+dy*(p1.y-cy));
477 Real c_const=(p1.x-cx)*(p1.x-cx)+(p1.y-cy)*(p1.y-cy)-r*r;
478 Real discriminant=b*b-4*a*c_const;
479 vector<Point> intersections;
480 if(almostEqual(discriminant,0)) {
481 Real t=-b/(2*a);
482 Real ix=p1.x+t*dx;
483 Real iy=p1.y+t*dy;
484 intersections.emplace_back(ix,iy);
485 intersections.emplace_back(ix,iy);
486 } else if(discriminant>0) {
487 Real sqrt_discriminant=sqrt(discriminant);
488 Real t1=(-b+sqrt_discriminant)/(2*a);
489 Real t2=(-b-sqrt_discriminant)/(2*a);
490 Real ix1=p1.x+t1*dx;
491 Real iy1=p1.y+t1*dy;
492 Real ix2=p1.x+t2*dx;
493 Real iy2=p1.y+t2*dy;
494 intersections.emplace_back(ix1,iy1);
495 intersections.emplace_back(ix2,iy2);
496 }
497 if(almostEqual(intersections[0].x,intersections[1].x)) {
498 if(greaterThan(intersections[0].y,intersections[1].y)) swap(intersections[0],intersections[1]);
499 } else if(greaterThan(intersections[0].x,intersections[1].x)) {
500 swap(intersections[0],intersections[1]);
501 }
502 return intersections;
503 }

509 Real d=sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
510 if(d>r1+r2 || d<abs(r1-r2)) return{};//No intersection
511 Real a=(r1*r1-r2*r2+d*d)/(2*d);
512 Real h=sqrt(r1*r1-a*a);
513 Real x0=x1+a*(x2-x1)/d;
514 Real y0=y1+a*(y2-y1)/d;
515 Real rx=-(y2-y1)*(h/d);
516 Real ry=(x2-x1)*(h/d);
517 Point p1(x0+rx,y0+ry);
518 Point p2(x0-rx,y0-ry);
519 vector<Point> intersections;
520 intersections.push_back(p1);
521 intersections.push_back(p2);
522 if(almostEqual(intersections[0].x,intersections[1].x)) {
523 if(greaterThan(intersections[0].y,intersections[1].y)) swap(intersections[0],intersections[1]);
524 } else if(greaterThan(intersections[0].x,intersections[1].x)) {
525 swap(intersections[0],intersections[1]);
526 }
527 return intersections;
528 }

530 /// @brief 点から引ける円の接線の接点を求める
531 vector<Point> getTangentLinesFromPoint(const Circle& c, const Point& p) {
532 Real cx=c.center.x,cy=c.center.y,r=c.r;
533 Real px=p.x,py=p.y;
534 Real dx=px-cx;
535 Real dy=py-cy;
536 Real d=(p-c.center).abs();
537 if(lessThan(d,r)) return {};//No tangents if the point is inside the circle
538 else if(almostEqual(d,r)) return {p};
539 Real a=r*r/d;
540 Real h=sqrt(r*r-a*a);
541 Real cx1=cx+a*dx/d;
542 Real cy1=cy+a*dy/d;
543 vector<Point> tangents;
544 tangents.emplace_back(cx1+h*dy/d,cy1-h*dx/d);
545 tangents.emplace_back(cx1-h*dy/d,cy1+h*dx/d);
546 if(almostEqual(tangents[0].x,tangents[1].x)) {
547 if(greaterThan(tangents[0].y,tangents[1].y)) swap(tangents[0],tangents[1]);
548 } else if(greaterThan(tangents[0].x,tangents[1].x)) {
549 swap(tangents[0],tangents[1]);
550 }
551 return tangents;
552 }

554 /// @brief 2つの円の共通接線を求める
555 vector<Segment> getCommonTangentsLine(const Circle& c1, const Circle& c2) {
556 Real x1=c1.center.x,y1=c1.center.y,r1=c1.r;
557 Real x2=c2.center.x,y2=c2.center.y,r2=c2.r;
558 Real dx=x2-x1;
559 Real dy=y2-y1;
560 Real d=sqrt(dx*dx+dy*dy);
561 vector<Segment> tangents;
562 //Coincident circles(infinite tangents)
563 if(almostEqual(d,0) && almostEqual(r1,r2)) return tangents;
564 //External tangents
565 if(greaterThanOrEqual(d,r1+r2)) {
566 Real a=atan2(dy,dx);
567 for(int sign: {-1,1}) {
568 Real theta=acos((r1+r2)/d);
569 Real cx1=x1+r1*cos(a+sign*theta);
570 Real cy1=y1+r1*sin(a+sign*theta);
571 Real cx2=x2+r2*cos(a+sign*theta);
572 Real cy2=y2+r2*sin(a+sign*theta);
573 tangents.emplace_back(Segment{Point(cx1,cy1),Point(cx2,cy2)});
574 if(almostEqual(d,r1+r2)) break;
575 }
576 }
577 //Internal tangents
578 if(greaterThanOrEqual(d,fabs(r1-r2))) {
579 Real a=atan2(dy,dx);
580 for(int sign: {-1,1}) {
581 Real theta=acos((r1-r2)/d);
582 Real cx1=x1+r1*cos(a+sign*theta);
583 Real cy1=y1+r1*sin(a+sign*theta);
584 Real cx2=x2-r2*cos(a+sign*theta);
585 Real cy2=y2-r2*sin(a+sign*theta);
586 tangents.emplace_back(Segment{Point(cx1,cy1),Point(cx2,cy2)});
587 if(almostEqual(d,fabs(r1-r2))) break;
588 }
589 }
590 sort(tangents.begin(),tangents.end(),[&](const Segment& s1, const Segment& s2) {
591 if(almostEqual(s1.a.x,s2.a.x)) return lessThan(s1.a.y,s2.a.y);
592 else return lessThan(s1.a.x,s2.a.x);
593 });
594 return tangents;
595 }
