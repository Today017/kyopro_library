47struct FastLCA {
48 ii n;
49 vi depth;
50 vi first_visit;
51 // st[i][j] : 長さ 2^i の区間における {深さ, 頂点番号} の最小値
52 vector<vector<pair<ii, ii>>> st;
53 vi log_table;
54 FastLCA()=default;
55
56 // adj は 0-indexed の隣接リスト、root は根の頂点番号
57 FastLCA(const vector<vi>& adj, ii root = 0) {
58 n = adj.size();
59 depth.assign(n, 0);
60 first_visit.assign(n, -1);
61 vector<pair<ii, ii>> euler_tour;
62 euler_tour.reserve(2 * n);
63
64 // オイラーツアーの構築
65 auto dfs = [&](auto& self, ii u, ii p, ii d) -> void {
66 first_visit[u] = euler_tour.size();
67 euler_tour.push_back({d, u});
68 depth[u] = d;
69 for (ii v : adj[u]) {
70 if (v != p) {
71 self(self, v, u, d + 1);
72 // 子から戻ってきたときにも追加
73 euler_tour.push_back({d, u});
74 }
75 }
76 };
77 dfs(dfs, root, -1, 0);
78
79 ii m = euler_tour.size();
80
81 // Sparse Table 用の対数テーブル事前計算
82 log_table.assign(m + 1, 0);
83 for (ii i = 2; i <= m; i++) {
84 log_table[i] = log_table[i >> 1] + 1;
85 }
86
87 // Sparse Table の構築
88 ii k = log_table[m] + 1;
89 st.assign(k, vector<pair<ii, ii>>(m));
90 for (ii i = 0; i < m; i++) {
91 st[0][i] = euler_tour[i];
92 }
93
94 for (ii j = 1; j < k; j++) {
95 for (ii i = 0; i + (1 << j) <= m; i++) {
96 st[j][i] = min(st[j - 1][i], st[j - 1][i + (1 << (j - 1))]);
97 }
98 }
99 }
100
101 // u と v の LCA を O(1) で取得
102 ii lca(ii u, ii v) const {
103 ii l = first_visit[u];
104 ii r = first_visit[v];
105 if (l > r) swap(l, r);
106
107 ii j = log_table[r - l + 1];
108 // 2つの区間の最小値を取る (区間は被っていてもよい性質を利用)
109 return min(st[j][l], st[j][r - (1 << j) + 1]).second;
110 }
111
112 // u と v の距離を O(1) で取得
113 ii dist(ii u, ii v) const {
114 return depth[u] + depth[v] - 2 * depth[lca(u, v)];
115 }
116};
