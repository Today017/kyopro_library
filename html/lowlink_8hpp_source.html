20 auto dfs = [&](auto&& dfs, ii now, ii pre, ii& cnt) -> void {
21 seen[now] = true;
22 ord[now] = low[now] = cnt++;
23 bool is_articulation = false;
24 ii child = 0;

26 if(!seen[e.to]) {
27 child++;

31 if(ord[now] < low[e.to]) bridge.push_back(minmax(now, e.to));
32 } else if(e.to != pre) low[now] = min(low[now], ord[e.to]);
33 }
34 if(pre == -1 && child > 1) is_articulation = true;
35 if(is_articulation) articulation.push_back(now);
36 };
37
38 ii cnt = 0;
39 rep(i, N) if(!seen[i]) dfs(dfs, i, -1, cnt);
40 sort(all(bridge));
41 sort(all(articulation));
