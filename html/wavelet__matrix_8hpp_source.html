4struct SuccinctBitVector {
5 static const ull NOTFOUND=0xFFFFFFFFFFFFFFFFLLU;
6
7private:
8 const ull size;
9 static const ull blockBitNum=16;
10 static const ull LEVEL_L=512;
11 static const ull LEVEL_S=16;
12 vector<ull> L;
13 vector<unsigned short> S;
14 vector<unsigned short> B;
15 ull num_one=0;
16
17public:
18 explicit SuccinctBitVector(const ull n) : size(n) {
19 const ull s=(n+blockBitNum-1)/blockBitNum+1;
20 this->B.assign(s,0);
21 this->L.assign(n/LEVEL_L+1,0);
22 this->S.assign(n/LEVEL_S+1,0);
23 }
24 void set_bit(const ull bit, const ull pos) {
25 assert(bit==0 or bit==1);
26 assert(pos<this->size);
27 const ull blockPos=pos/blockBitNum;
28 const ull offset=pos%blockBitNum;
29 if(bit==1) B.at(blockPos)|=(1LLU<<offset);
30 else B.at(blockPos)&=(~(1LLU<<offset));
31 }
32 ull access(const ull pos) {
33 assert(pos<this->size);
34 const ull blockPos=pos/blockBitNum;
35 const ull offset=pos%blockBitNum;
36 return((B.at(blockPos)>>offset)&1);
37 }
38 void build() {
39 ull num=0;
40 for(ull i=0; i<=size; i++) {
41 if(i%LEVEL_L==0) L.at(i/LEVEL_L)=num;
42 if(i%LEVEL_S==0) S.at(i/LEVEL_S)=num-L.at(i/LEVEL_L);
43 if(i!=size && i%blockBitNum==0) num+=this->pop_count(this->B.at(i/blockBitNum));
44 }
45 this->num_one=num;
46 }
47 ull rank(const ull bit, const ull pos) {
48 assert(bit==0 or bit==1);
49 assert(pos<=this->size);
50 if(bit) return L[pos/LEVEL_L]+S[pos/LEVEL_S]+pop_count(B[pos/blockBitNum]&((1<<(pos%blockBitNum))-1));
51 else return pos-rank(1,pos);
52 }
53 ull select(const ull bit, const ull rank) {
54 assert(bit==0 or bit==1);
55 assert(rank>0);
56 if(bit==0 && rank>this->size-this->num_one) return NOTFOUND;
57 if(bit==1 && rank>this->num_one) return NOTFOUND;
58 ull large_idx=0;
59 {
60 ull left=0;
61 ull right=L.size();
62 while(right-left>1) {
63 ull mid=(left+right)/2;
64 ull r=L.at(mid);
65 r=(bit)? r: mid*LEVEL_L-L.at(mid);
66 if(r<rank) {
67 left=mid;
68 large_idx=mid;
69 } else {
70 right=mid;
71 }
72 }
73 }
74 ull small_idx=(large_idx*LEVEL_L)/LEVEL_S;
75 {
76 ull left=(large_idx*LEVEL_L)/LEVEL_S;
77 ull right=min(((large_idx+1)*LEVEL_L)/LEVEL_S,(ull)S.size());
78 while(right-left>1) {
79 ull mid=(left+right)/2;
80 ull r=L.at(large_idx)+S.at(mid);
81 r=(bit)? r: mid*LEVEL_S-r;
82 if(r<rank) {
83 left=mid;
84 small_idx=mid;
85 } else {
86 right=mid;
87 }
88 }
89 }
90 ull rank_pos=0;
91 {
92 const ull begin_block_idx=(small_idx*LEVEL_S)/blockBitNum;
93 ull total_bit=L.at(large_idx)+S.at(small_idx);
94 if(bit==0) total_bit=small_idx*LEVEL_S-total_bit;
95 for(ull i=0;;++i) {
96 ull b=pop_count(B.at(begin_block_idx+i));
97 if(bit==0)b=blockBitNum-b;
98 if(total_bit+b>=rank) {
99 ull block=(bit)? B.at(begin_block_idx+i): ~B.at(begin_block_idx+i);
100 rank_pos=(begin_block_idx+i)*blockBitNum+select_in_block(block,rank-total_bit);
101 break;
102 }
103 total_bit+=b;
104 }
105 }
106 return rank_pos+1;
107 }
108 ull get_num_one() const {
109 return num_one;
110 }
111
112private:
113 ull pop_count(ull x) {
114 x=(x&0x5555555555555555ULL)+((x>>1)&0x5555555555555555ULL);
115 x=(x&0x3333333333333333ULL)+((x>>2)&0x3333333333333333ULL);
116 x=(x+(x>>4))&0x0f0f0f0f0f0f0f0fULL;
117 x=x+(x>>8);
118 x=x+(x>>16);
119 x=x+(x>>32);
120 return x&0x7FLLU;
121 }
122 ull select_in_block(ull x, ull rank) {
123 ull x1=x-((x&0xAAAAAAAAAAAAAAAALLU)>>1);
124 ull x2=(x1&0x3333333333333333LLU)+((x1>>2)&0x3333333333333333LLU);
125 ull x3=(x2+(x2>>4))&0x0F0F0F0F0F0F0F0FLLU;
126 ull pos=0;
127 for(;;pos+=8) {
128 ull rank_next=(x3>>pos)&0xFFLLU;
129 if(rank<=rank_next) break;
130 rank-=rank_next;
131 }
132 ull v2=(x2>>pos)&0xFLLU;
133 if(rank>v2) {
134 rank-=v2;
135 pos+=4;
136 }
137 ull v1=(x1>>pos)&0x3LLU;
138 if(rank>v1) {
139 rank-=v1;
140 pos+=2;
141 }
142 ull v0=(x>>pos)&0x1LLU;
143 if(v0<rank) {
144 rank-=v0;
145 pos+=1;
146 }
147 return pos;
148 }

167 ///@brief コンストラクタ
168 ///@brief a から Wavelet Matrix を構築する
169 WaveletMatrix(const vector<ull>&a) {
170 assert(a.size()>0);
171 size=a.size();
172 maximum_element=*max_element(all(a))+1;
173 bit_size=get_num_of_bit(maximum_element);
174 if(bit_size==0) bit_size=1;
175 for(ull i=0; i<bit_size; ++i) {
176 SuccinctBitVector sv(size);
177 bit_arrays.push_back(sv);
178 }
179 this->begin_one.resize(bit_size);
180 this->cumulative_sum.resize(bit_size+1,vector<ull>(size+1,0));
181 for(ull j=0; j<a.size(); ++j) {
182 this->cumulative_sum.at(0).at(j+1)=this->cumulative_sum.at(0).at(j)+a[j];
183 }
184 vector<ull> v(a);
185 for(ull i=0; i<bit_size; ++i) {
186 vector<ull>temp;
187 for(ull j=0; j<v.size(); ++j) {
188 ull c=v.at(j);
189 ull bit=(c>>(bit_size-i-1))&1;
190 if(bit==0) {
191 temp.push_back(c);
192 bit_arrays.at(i).set_bit(0,j);
193 }
194 }
195 this->begin_one.at(i)=temp.size();
196 for(ull j=0; j<v.size(); ++j) {
197 ull c=v.at(j);
198 ull bit=(c>>(bit_size-i-1))&1;
199 if(bit==1) {
200 temp.push_back(c);
201 bit_arrays.at(i).set_bit(1,j);
202 }
203 }
204 for(ull j=0; j<temp.size(); ++j) {
205 this->cumulative_sum.at(i+1).at(j+1)=this->cumulative_sum.at(i+1).at(j)+temp.at(j);
206 }
207 bit_arrays.at(i).build();
208 v=temp;
209 }
210 for(int i=v.size()-1; i>=0; --i) this->begin_alphabet[v.at(i)]=i;
211 }

255 ///@brief 区間 [l, r) で k 番目に小さい値のインデックスを返す
256 ///@brief k は 0-indexed
257 ///@note O(log σ)
258 ull quantile_range(ull begin_pos, ull end_pos, ull k) {
259 if((end_pos>size||begin_pos>=end_pos)||(k>=end_pos-begin_pos)) return NOTFOUND;
260 ull val=0;
261 for(ull i=0; i<bit_size; ++i) {
262 const ull num_of_zero_begin=bit_arrays.at(i).rank(0,begin_pos);
263 const ull num_of_zero_end=bit_arrays.at(i).rank(0,end_pos);
264 const ull num_of_zero=num_of_zero_end-num_of_zero_begin;
265 const ull bit=(k<num_of_zero)?0:1;
266 if(bit) {
267 k-=num_of_zero;
268 begin_pos=this->begin_one.at(i)+begin_pos-num_of_zero_begin;
269 end_pos=this->begin_one.at(i)+end_pos-num_of_zero_end;
270 } else {
271 begin_pos=num_of_zero_begin;
272 end_pos=num_of_zero_begin+num_of_zero;
273 }
274 val=((val<<1)|bit);
275 }
276 ull left=0;
277 for(ull i=0; i<bit_size; ++i) {
278 const ull bit=(val>>(bit_size-i-1))&1;
279 left=bit_arrays.at(i).rank(bit,left);
280 if(bit) left+=this->begin_one.at(i);
281 }
282 const ull rank=begin_pos+k-left+1;
283 return select(val,rank)-1;
284 }

326 ///@brief 区間 [l, r) の (c と同じ値の数, c より小さい値の数, c より大きい値の数)
327 ///@note O(log σ)
328 tuple<ull,ull,ull>rank_all(const ull c, ull begin, ull end) {
329 assert(end<=size);
330 const ull num=end-begin;
331 if(begin>=end) return make_tuple(0,0,0);
332 if(c>=maximum_element || end==0) return make_tuple(0,num,0);
333 ull rank_less_than=0,rank_more_than=0;
334 for(size_t i=0; i<bit_size && begin<end; ++i) {
335 const ull bit=(c>>(bit_size-i-1))&1;
336 const ull rank0_begin=this->bit_arrays.at(i).rank(0,begin);
337 const ull rank0_end=this->bit_arrays.at(i).rank(0,end);
338 const ull rank1_begin=begin-rank0_begin;
339 const ull rank1_end=end-rank0_end;
340 if(bit) {
341 rank_less_than+=(rank0_end-rank0_begin);
342 begin=this->begin_one.at(i)+rank1_begin;
343 end=this->begin_one.at(i)+rank1_end;
344 } else {
345 rank_more_than+=(rank1_end-rank1_begin);
346 begin=rank0_begin;
347 end=rank0_end;
348 }
349 }
350 const ull rank=num-rank_less_than-rank_more_than;
351 return make_tuple(rank,rank_less_than,rank_more_than);
352 }

354 ///@brief 区間 [l, r) で出現回数が多い順に k 個の (値, 頻度) を返す
355 ///@note O(log σ)
356 vector<pair<ull,ull>>topk(ull begin,ull end,ull k) {
357 ull s=begin,e=end;
358 assert(s<e);
359 vector<pair<ull,ull>>result;
360 auto c=[](const tuple<ull,ull,ull,ull,ull>&l,const tuple<ull,ull,ull,ull,ull>&r) {
361 if(get<0>(l)!=get<0>(r))return get<0>(l)<get<0>(r);
362 if(get<3>(l)!=get<3>(r))return get<3>(l)>get<3>(r);
363 if(get<4>(l)!=get<4>(r))return get<4>(l)>get<4>(r);
364 return true;
365 };
366 std::priority_queue<tuple<ull,ull,ull,ull,ull>,vector<tuple<ull,ull,ull,ull,ull>>,decltype(c)>que(c);//width,left,right,depth,value
367 que.push(make_tuple(e-s,s,e,0,0));
368 while(not que.empty()) {
369 auto element=que.top();
370 que.pop();
371 ull width,left,right,depth,value;
372 tie(width,left,right,depth,value)=element;
373 if(depth>=this->bit_size) {
374 result.emplace_back(make_pair(value,right-left));
375 if(result.size()>=k)break;
376 continue;
377 }
378 const ull left0=this->bit_arrays.at(depth).rank(0,left);
379 const ull right0=this->bit_arrays.at(depth).rank(0,right);
380 if(left0<right0)que.push(make_tuple(right0-left0,left0,right0,depth+1,value));
381 const ull left1=this->begin_one.at(depth)+this->bit_arrays.at(depth).rank(1,left);
382 const ull right1=this->begin_one.at(depth)+this->bit_arrays.at(depth).rank(1,right);
383 if(left1<right1)que.push(make_tuple(right1-left1,left1,right1,depth+1,value|(1<<(bit_size-depth-1))));
384 }
385 return result;
386 };

516 ull range_sum(const ull begin, const ull end, const ull depth, const ull c, const ull x, const ull y) {
517 if(begin==end)return 0;
518 if(depth==bit_size) {
519 if(x<=c and c<y) return c*(end-begin);
520 return 0;
521 }
522 const ull next_c=((ull)1<<(bit_size-depth-1))|c;
523 const ull all_one_c=(((ull)1<<(bit_size-depth-1))-1)|next_c;
524 if(all_one_c<x or y<=c)return 0;
525 if(x<=c and all_one_c<y) return this->cumulative_sum.at(depth).at(end)-this->cumulative_sum.at(depth).at(begin);
526 const ull rank0_begin=this->bit_arrays.at(depth).rank(0,begin);
